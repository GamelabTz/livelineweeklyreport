-- Create reports table
CREATE TABLE reports (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    line VARCHAR(255) NOT NULL,
    from_person VARCHAR(255) NOT NULL,
    to_person VARCHAR(255) NOT NULL,
    report_date DATE NOT NULL,
    team VARCHAR(255) NOT NULL,
    location VARCHAR(255) NOT NULL,
    ref VARCHAR(255) NOT NULL,
    status VARCHAR(20) DEFAULT 'draft',
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Create work_days table
CREATE TABLE work_days (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    report_id UUID REFERENCES reports(id) ON DELETE CASCADE,
    day_name VARCHAR(20) NOT NULL,
    work_date DATE NOT NULL,
    work_type VARCHAR(20) NOT NULL,
    no_work_reason VARCHAR(255),
    no_work_details TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Create towers table
CREATE TABLE towers (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    work_day_id UUID REFERENCES work_days(id) ON DELETE CASCADE,
    tower_number INTEGER NOT NULL,
    tower_type VARCHAR(10) NOT NULL,
    insulators_r INTEGER DEFAULT 0,
    insulators_y INTEGER DEFAULT 0,
    insulators_b INTEGER DEFAULT 0,
    remarks TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Create images table
CREATE TABLE images (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    work_day_id UUID REFERENCES work_days(id) ON DELETE CASCADE,
    image_url TEXT NOT NULL,
    caption TEXT,
    file_name VARCHAR(255),
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Create lines table for storing transmission lines
CREATE TABLE transmission_lines (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    name VARCHAR(255) NOT NULL UNIQUE,
    voltage INTEGER NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

CREATE POLICY "Enable all operations for authenticated users" ON images

-- Create a separate index for search performance
CREATE INDEX idx_weekly_reports_persons ON weekly_reports(from_person, to_person);

-- Create the table with additional constraints and optimizations
CREATE TABLE weekly_reports (
    id BIGINT PRIMARY KEY GENERATED ALWAYS AS IDENTITY,
    line TEXT NOT NULL CHECK (length(line) <= 500),
    from_person UUID REFERENCES auth.users(id) ON DELETE RESTRICT,
    to_person UUID REFERENCES auth.users(id) ON DELETE RESTRICT,
    report_date DATE NOT NULL CHECK (report_date <= CURRENT_DATE),
    team TEXT NOT NULL CHECK (length(team) >= 2 AND length(team) <= 100),
    location TEXT NOT NULL CHECK (length(location) >= 2 AND length(location) <= 200),
    reference TEXT CHECK (reference IS NULL OR length(reference) <= 255),
    work_days JSONB NOT NULL 
        CHECK (jsonb_typeof(work_days) = 'object' AND jsonb_array_length(work_days::jsonb) > 0),
    status TEXT DEFAULT 'draft' 
        CHECK (status IN ('draft', 'submitted', 'approved', 'rejected')),
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW() NOT NULL,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW() NOT NULL
);

-- Create trigger for automatic updated_at timestamp
CREATE OR REPLACE FUNCTION update_modified_column()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = NOW();
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER update_weekly_reports_modtime
BEFORE UPDATE ON weekly_reports
FOR EACH ROW
EXECUTE FUNCTION update_modified_column();

-- Enable RLS
ALTER TABLE weekly_reports ENABLE ROW LEVEL SECURITY;

-- Comprehensive RLS policies
CREATE POLICY "Users can view related reports" ON weekly_reports
FOR SELECT 
TO authenticated
USING (
    (SELECT auth.uid()) = from_person 
    OR (SELECT auth.uid()) = to_person
);

CREATE POLICY "Users can insert own reports" ON weekly_reports
FOR INSERT 
TO authenticated
WITH CHECK (
    (SELECT auth.uid()) = from_person 
    AND status = 'draft'
);

CREATE POLICY "Users can update own draft reports" ON weekly_reports
FOR UPDATE 
TO authenticated
USING (
    (SELECT auth.uid()) = from_person 
    AND status = 'draft'
);

CREATE POLICY "Team leads can approve reports" ON weekly_reports
FOR UPDATE 
TO authenticated
USING (
    (SELECT auth.uid()) = to_person 
    AND status IN ('draft', 'submitted')
)
WITH CHECK (
    status IN ('approved', 'rejected')
);
